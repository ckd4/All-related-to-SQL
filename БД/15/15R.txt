SELECT DISTINCT Номер_лаборатории FROM Расписание WHERE День_недели LIKE 'Понедельник' AND Номер_пары  = 2

--2--
SELECT DISTINCT Номер_группы FROM Расписание
WHERE День_недели LIKE '%Среда%' AND Номер_лаборатории = 106 AND Номер_группы LIKE '%2-%'

--3--
SELECT [Номер_группы], [Номер_лаборатории], [Дисциплина], [День_недели], [Номер_пары] INTO задание3 FROM Расписание 
WHERE День_недели LIKE 'Понедельник' AND Номер_пары = 2

--4--
INSERT INTO задание3 (Номер_группы, Номер_лаборатории, Дисциплина, День_недели, Номер_пары) SELECT Номер_группы, Номер_лаборатории, Дисциплина, День_недели, Номер_пары FROM Расписание 
WHERE Номер_группы LIKE '%1-%' AND NOT EXISTS(SELECT DISTINCT Номер_группы FROM задание3 
WHERE Номер_группы LIKE '%1-%')

--5--
UPDATE задание3 SET Номер_пары = Номер_пары + 1 
WHERE (День_недели LIKE 'Вторник' OR День_недели LIKE 'Четверг' OR День_недели LIKE 'Суббота')

--6--
DELETE FROM задание3 
WHERE Номер_лаборатории = 106 AND День_недели LIKE 'Понедельник'

--7--
SELECT distinct Номер_лаборатории, COUNT(distinct Номер_группы) AS Количество_занятий 
FROM Расписание GROUP BY Номер_лаборатории
