--1--
SELECT *
FROM dbo.Книги Код_автора
WHERE Код_автора.Год_издания BETWEEN 1990 AND 1999 AND Количество_страниц > 200

--2--	
SELECT *
FROM dbo.Авторы Код_автора
WHERE Фамилия LIKE 'В%' AND Национальность = 'Русский'

--3--
SELECT Книги.Код_книги, Книги.Название, Книги.Год_издания, Книги.Количество_страниц, Книги.Издательство, Авторы.Фамилия, Авторы.Имя, Авторы.Отчество, Авторы.Код_автора
into таблица
FROM Авторы INNER join Книги ON Авторы.Код_автора = Книги.Код_книги
WHERE Год_издания BETWEEN 1990 AND 1999 AND Количество_страниц > 200

--4--
INSERT таблица(Код_книги, Авторы.Код_автора,Название,Год_издания,Количество_страниц, Имя, Фамилия, Отчество)
SELECT Код_книги,Авторы.Код_автора,Название,Год_издания,Количество_страниц, Имя, Фамилия, Отчество
FROM Авторы INNER join Книги ON Авторы.Код_автора = Книги.Код_книги
WHERE Год_издания between 1980 and 1989
 
--5--
UPDATE таблица
SET Год_издания = Год_издания + 1
WHERE Год_издания=1995 

--6--
delete from таблица
where Фамилия like 'А%'

--7--
select Фамилия,
count(Фамилия) as Количество_книг from Авторы group by Фамилия
